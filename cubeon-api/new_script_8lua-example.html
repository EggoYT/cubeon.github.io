This is an example of creating script for blocks

// Warning! Links on documentation can work wrong. Be careful.
blocks.example = {} // as an "example" means real block name.
blocks.example.type = "cubeon.example" // Here you need to put which type of blocks this script will process.
// Local variables needs to init some information unique for this script.
// Don't create global variables, it's can breake system working.
local block;
local param1;
local param2;
// This is an init function. Here you can prepare your local variables for future use in tick function.
// In example, here, I get Block object of this script, and get it's static_params, which means constants.
function blocks.example.init()
block = World.Blocks.getBlockByName(blocks.example.type)
param1 = block.static_params["param1"]:getInt();
param2 = block.static_params["param2"]:getString();
end;
// This is an tick function. It's calls every tick (20 times per second) for each loaded Chunk and each "active" Block type, placed into chunk.
// If Chunk haven't placed blocks of this type, this function will not call for this chunk. 1 chunk - 1 call per tick
// here "x" and "z" means coordinates of current Chunk, whose blocks are passed into "arr" as BlockArray
// "x" and "z" have int types
function blocks.example.tick(x, z, arr)
// And here you can use documented functions.
// As an example, I break specific block under those type block
for i = 0, arr.size - 1 do // Basic cycle for all blocks in chunk, placed in BlockArray
blockParams = arr:get(i); // Get current BlockParams object
underBlock = World.getBlock(blockParams.x + x * 16, blockParams.y - 1, blockParams.z + z * 16); // Get block under ours. Don't forget add chunk coords to block!
if (underBlock ~= 0) and (World.Blocks.getBlockById(underBlock).name == "cubeon.someBlock") then // Check block under those. If block is that, what we need, then remove it.
World.setBlock(blockParams.x + x * 16, blockParams.y - 1, blockParams.z + z * 16, 0); // 0 type is Air
end;
end;
end;
