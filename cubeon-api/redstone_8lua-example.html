blocks["cubeon.redstone"] = {}
local namespace = blocks["cubeon.redstone"]
local update_nearest
local update
local get_signal
local check_redstone
local is_block_conductor
local types = {
source = 0,
receiver = 1,
transmitter = 2
}
check_redstone = function(x, y, z)
local block, block_state, block_struct = unpack(World.get_block_data(x, y, z))
if block_struct == nil then
return nil
end
if not block:in_group("cubeon.redstone") then
return nil
end
return block, block_struct
end
is_block_conductor = function(x, y, z)
return World.get_state(x, y, z):is_box_full()
end
local raw_get_signal = function(x, y, z, offset)
local block, block_struct
local block_space
block, block_struct = check_redstone(x + offset.x, y + offset.y, z + offset.z)
if block == nil then
return nil
end
if block.static_params["channel_type"] == types.receiver then --it can't share for others
return nil
end
if block_struct.params["source"] == -offset then
return nil
end
block_space = blocks[block.name]
if block_space ~= nil and block_space.get_signal ~= nil then
status, msg = pcall(block_space.get_signal, block_struct, x + offset.x, y + offset.y, z + offset.z, offset)
if not status then
print("can't turning block " .. block.name .. " caused to " .. msg)
else
return msg - 1
end
end
return block_struct.params["signal"] - 1
end
get_signal = function(x, y, z, offset)
local got_signal
local yo
for yo = -1, 1 do
if not((yo == -1 and is_block_conductor(x + offset.x, y, z + offset.z)) or (yo == 1 and is_block_conductor(x, y + 1, z))) then
got_signal = raw_get_signal(x, y, z, offset + MathAPI.ivec3(0, yo, 0))
if got_signal ~= nil then
return got_signal, yo
end
end
end
return 0, 0
end
update_nearest = function(x, y, z, non_checking)
local offset
local block_struct
local block
local yo
offset = MathAPI.ivec3(0, 1, 0)
if non_checking ~= offset then
block_struct = World.get_block_struct(x, y + 1, z)
if block_struct ~= nil then
block = block_struct:get_block_object()
if block:in_group("cubeon.redstone") then
update(x, y + 1, z, block, block_struct, -offset)
end
end
end
local yo
for yo = -1, 1 do
offset = MathAPI.ivec3(1, yo, 0)
if non_checking ~= offset then
block_struct = World.get_block_struct(x + 1, y + yo, z)
if block_struct ~= nil then
block = block_struct:get_block_object()
if block:in_group("cubeon.redstone") then
update(x + 1, y + yo, z, block, block_struct, -offset)
end
end
end
offset = MathAPI.ivec3(-1, yo, 0)
if non_checking ~= offset then
block_struct = World.get_block_struct(x - 1, y + yo, z)
if block_struct ~= nil then
block = block_struct:get_block_object()
if block:in_group("cubeon.redstone") then
update(x - 1, y + yo, z, block, block_struct, -offset)
end
end
end
offset = MathAPI.ivec3(0, yo, -1)
if non_checking ~= offset then
block_struct = World.get_block_struct(x, y + yo, z - 1)
if block_struct ~= nil then
block = block_struct:get_block_object()
if block:in_group("cubeon.redstone") then
update(x, y + yo, z - 1, block, block_struct, -offset)
end
end
end
offset = MathAPI.ivec3(0, yo, 1)
if non_checking ~= offset then
block_struct = World.get_block_struct(x, y + yo, z + 1)
if block_struct ~= nil then
block = block_struct:get_block_object()
if block:in_group("cubeon.redstone") then
update(x, y + yo, z + 1, block, block_struct, -offset)
end
end
end
end
end
update = function(x, y, z, block, block_struct)
if block.static_params["channel_type"] == types.source or block_struct.params["signal"] == -1 then
return
end
local block_space = blocks[block.name]
local b_signal = 0
local b_offset
if block_space ~= nil and block_space.ask_signal ~= nil then
local status
status, b_signal, b_offset = pcall(block_space.ask_signal, x, y, z, get_signal)
if not status then
print("can't take signal from script " .. block.name .. " caused to " .. b_signal)
b_signal = 0
end
else
local c_offset
local c_signal
c_offset = MathAPI.ivec3(1, 0, 0)
c_signal, c_offset.y = get_signal(x, y, z, c_offset)
if c_signal > b_signal then
b_signal = c_signal
b_offset = c_offset
end
c_offset = MathAPI.ivec3(-1, 0, 0)
c_signal, c_offset.y = get_signal(x, y, z, c_offset)
if c_signal > b_signal then
b_signal = c_signal
b_offset = c_offset
end
c_offset = MathAPI.ivec3(0, 0, 1)
c_signal, c_offset.y = get_signal(x, y, z, c_offset)
if c_signal > b_signal then
b_signal = c_signal
b_offset = c_offset
end
c_offset = MathAPI.ivec3(0, 0, -1)
c_signal, c_offset.y = get_signal(x, y, z, c_offset)
if c_signal > b_signal then
b_signal = c_signal
b_offset = c_offset
end
c_offset = MathAPI.ivec3(0, -1, 0)
c_signal = raw_get_signal(x, y, z, c_offset)
if c_signal ~= nil and c_signal > b_signal then
b_signal = c_signal
b_offset = c_offset
end
end
local current_signal = block_struct.params["signal"]
if b_signal ~= current_signal or b_offset ~= block_struct.params["source"] then
block_struct.params["signal"] = b_signal
block_struct.params["source"] = b_offset
if channel_type ~= types.source and block_space ~= nil and block_space.update_signal ~= nil then
local status, updating_nearest = pcall(block_space.update_signal, block, block_struct, MathAPI.lvec3(x, y, z), current_signal)
if not status then
print("can't update signal for block " .. block.name .. " caused to " .. update_nearest)
updating_nearest = false
end
end
if updating_nearest ~= true then
update_nearest(x, y, z, b_offset)
end
end
end
function namespace.is_signal(x, y, z, offset)
local yo
for yo = -1, 1 do
if not (yo == -1 and World.is_block(x + offset.x, y, z + offset.z)) or not (yo == 1 and World.is_block(x, y + 1, z)) then
local block, block_struct = check_redstone(x, y + yo, z)
if block ~= nil then
local block_space = blocks[block.name]
if block_space ~= nil and block_space.is_signal ~= nil then
local status, msg = pcall(block_space.is_signal, x, y + yo, z, offset - MathAPI.ivec3(0, yo, 0), block, World.get_state(x, y + yo, z))
if not status then
print("can't check signal from method is_signal from block " .. block.name .. " caused to " .. msg)
return true
end
return msg, yo
else
return true, yo
end
end
end
end
return false, nil
end
function namespace.update_model(state, x, y, z, recurse)
local block_space = blocks[World.Blocks.getBlockById(state:get_id()).name]
if block_space ~= nil and block_space.update_model ~= nil then
local status, msg = pcall(block_space.update_model, state, x, y, z, recurse)
if not status then
print("can't check signal from method is_signal from block " .. block.name .. " caused to " .. msg)
end
end
end
function namespace.send_signal(coords, offset, signal)
block, block_struct = check_redstone(coords.x + offset.x, coords.y + offset.y, coords.z + offset.z)
if block ~= nil then
update(coords.x + offset.x, coords.y + offset.y, coords.z + offset.z, block, block_struct)
end
end
function namespace.on_create(block, block_coords, block_state, block_struct, face)
block_struct.params["signal"] = block.static_params["init_signal"]
if block.static_params["channel_type"] == types.source then
update_nearest(block_coords.x, block_coords.y, block_coords.z)
else
update(block_coords.x, block_coords.y, block_coords.z, block, block_struct)
end
end
function namespace.on_neighbor_changed(block, coords, state, struct, offset, near_state)
update(coords.x, coords.y, coords.z, block, struct)
update_nearest(coords.x, coords.y, coords.z)
end
function namespace.on_destroy(block, block_coords, block_state, block_struct, face)
block_struct.params["signal"] = -1
update_nearest(block_coords.x, block_coords.y, block_coords.z, block_struct.params["source"])
end
