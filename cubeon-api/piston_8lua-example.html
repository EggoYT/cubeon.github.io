blocks["cubeon.piston"] = {}
local namespace = blocks["cubeon.piston"]
local direction_list --i guess here must be only constants
function namespace.init()
direction_list = enums.faces.facial
end
calc_dir = function(dir)
local offset = MathAPI.lvec3(0, 0, 0)
if dir == direction_list.up then --check piston face direction
offset.y = 1
elseif dir == direction_list.down then
offset.y = -1
elseif dir == direction_list.north then
offset.z = -1
elseif dir == direction_list.south then
offset.z = 1
elseif dir == direction_list.east then
offset.x = 1
elseif dir == direction_list.west then
offset.x = -1
end
return offset
end
local function turn_on(block, block_struct, coords)
local block_state = World.get_state(coords.x, coords.y, coords.z)
if block_state:get_model_state_id() ~= 0 then
return
end
local offset = calc_dir(block_state:get_rotation())
local offseted_coords = coords + offset
World.replace_block(offseted_coords.x, offseted_coords.y, offseted_coords.z, coords.x + offset.x * 2, coords.y + offset.y * 2, coords.z + offset.z * 2) --replace block with piston
local events_id = World.set_block_deferred_events(offseted_coords.x, offseted_coords.y, offseted_coords.z, block.index)
if events_id == -1 then
return
end
block_state:set_model_state_id(2)
local addition_state = World.get_state(offseted_coords.x, offseted_coords.y, offseted_coords.z)
addition_state:set_model_state_id(1)
addition_state:set_rotation(block_state:get_rotation())
World.call_events(events_id)
block_struct.params["activated"] = true
end
local function turn_off(block, block_struct, coords)
if block_struct.params["activated"] ~= true then
return
end
local block_state = World.get_state(coords.x, coords.y, coords.z)
if block_state:get_model_state_id() == 1 then
return
end
block_state:set_model_state_id(0)
local offset = calc_dir(block_state:get_rotation())
local offseted_coords = coords + offset
World.set_block(offseted_coords.x, offseted_coords.y, offseted_coords.z, 0, 1) --удаляем блок из какого-либо скрипта, где 1 есть наш параметр
block_struct.params["activated"] = nil
end
function namespace.update_signal(block, block_struct, coords, old_signal)
if old_signal == 0 then
turn_on(block, block_struct, coords)
elseif block_struct.params["signal"] == 0 then
turn_off(block, block_struct, coords)
end
end
function namespace.on_destroy(block, coords, block_state, block_struct, face, event_type) --event_type и есть переданный параметр при set_block (равен 1)
if event_type == 1 then --broke by another piston
return
end
local state_id = block_state:get_model_state_id()
if state_id == 0 then
return
end
local offseted_coords
if state_id == 1 then
offseted_coords = coords - calc_dir(block_state:get_rotation())
else
offseted_coords = coords + calc_dir(block_state:get_rotation())
end
World.set_block(offseted_coords.x, offseted_coords.y, offseted_coords.z, 0, 1)
end
