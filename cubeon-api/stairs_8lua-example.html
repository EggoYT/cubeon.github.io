blocks["cubeon.stairs"] = {} --register script for this type of block/group of blocks
local namespace = blocks["cubeon.stairs"]
local directions
local calculate_rotation
local check_stairs
function namespace.init()
directions = enums.faces.multi_facial
end
check_stairs = function(x, y, z, block_state) --returning direction
local state = World.get_state(x, y, z)
if state:get_id() ~= block_state:get_id() then
return nil
end
if MathAPI.bit_xor(state:get_direction(), block_state:get_direction()) % 2 == 1 then
return MathAPI.bit_and(state:get_direction(), directions.xz_mask), MathAPI.bit_and(state:get_direction(), directions.down)
end
return nil
end
calculate_rotation = function(x, y, z, block_state, angled)
local took_rotation, uped_took = nil
local this_rotation = MathAPI.bit_and(block_state:get_direction(), directions.xz_mask)
local uped = MathAPI.bit_and(block_state:get_direction(), directions.down)
if this_rotation == directions.north then
took_rotation, uped_took = check_stairs(x, y, z - angled, block_state)
elseif this_rotation == directions.south then
took_rotation, uped_took = check_stairs(x, y, z + angled, block_state)
elseif this_rotation == directions.west then
took_rotation, uped_took = check_stairs(x - angled, y, z, block_state)
else
took_rotation, uped_took = check_stairs(x + angled, y, z, block_state)
end
if took_rotation ~= nil and uped_took == uped then
if uped ~= 0 then
if this_rotation == MathAPI.bit_and(took_rotation + 1, 3) then
this_rotation = took_rotation
end
block_state:set_rotation(this_rotation + uped)
else
if this_rotation ~= MathAPI.bit_and(took_rotation + 1, 3) then
this_rotation = took_rotation
end
block_state:set_rotation(this_rotation)
end
if angled == 1 then
block_state:set_model_state("out_angled")
else
block_state:set_model_state("in_angled")
end
elseif angled == -1 or not calculate_rotation(x, y, z, block_state, -1) then
block_state:set_model_state("_default")
block_state:set_rotation(MathAPI.bit_or(uped, this_rotation))
return false
end
return true
end
function namespace.on_create(block, block_coords, block_state, block_struct)
calculate_rotation(block_coords.x, block_coords.y, block_coords.z, block_state, 1)
end
function namespace.on_neighbor_changed(block, block_coords, block_state, block_struct, coords_offset)
calculate_rotation(block_coords.x, block_coords.y, block_coords.z, block_state, 1)
end
